USE [SchedarioTerritorio]
INSERT INTO [ProvvPagamentiAccentrati]
           (
           [TipologiaAccentramento]
           ,[Seprag]
           ,[NumeroFattura]
           ,[NumeroDocumento]
           ,[Data]
           ,[SepragEmittente]
           ,[Pagamento]
			)
			SELECT TOP 100 PERCENT
				   n.TIPOLOGIA_ACCENTRAMENTO
				  ,n.COD_SEPRAG
				  ,n.NUM_FATTURA
				  ,n.NUM_DOCUMENTO
				  ,n.DATA
				  ,n.SEPRAG_EMITTENTE
				  ,n.PAGAMENTO
			  FROM [_oraAccentramenti] AS n
			  LEFT JOIN [ProvvPagamentiAccentrati] AS m 
				ON 
				 n.TIPOLOGIA_ACCENTRAMENTO = m.TipologiaAccentramento AND
				 n.COD_SEPRAG = m.Seprag AND
				 n.NUM_FATTURA = m.NumeroFattura AND
				 n.NUM_DOCUMENTO = m.NumeroDocumento AND
				 n.DATA = m.Data AND
				 n.SEPRAG_EMITTENTE = m.SepragEmittente AND
				 n.PAGAMENTO = m.Pagamento				 
		     WHERE 
				 m.TipologiaAccentramento IS NULL AND
				 m.Seprag IS NULL AND
				 m.NumeroFattura IS NULL AND
				 m.NumeroDocumento IS NULL AND
				 m.Data IS NULL AND
				 m.SepragEmittente IS NULL AND
				 m.Pagamento IS NULL 
			 GROUP BY 
			   n.TIPOLOGIA_ACCENTRAMENTO
			  ,n.COD_SEPRAG
			  ,n.NUM_FATTURA
			  ,n.NUM_DOCUMENTO
			  ,n.DATA
			  ,n.SEPRAG_EMITTENTE
			  ,n.PAGAMENTO
